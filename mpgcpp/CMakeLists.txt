cmake_minimum_required(VERSION 3.24)
project(mpgcpp)

set(CMAKE_CXX_STANDARD 20)
set(HEADER_FILES MeanPayoffGame.h Variable.h MaxAtomSystem.h MinMaxSystem.h IntRange.h MPGReader.h Writer.h ProgramOptions.h)
set(SOURCE_FILES Variable.cpp Writer.cpp ProgramOptions.cpp)
include_directories(CPLibrary/include)
add_library(CPLibrary STATIC IMPORTED)
find_package(Boost COMPONENTS iostreams program_options log REQUIRED)
add_executable(mpgcpp main.cpp ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(mpgcpp Boost::iostreams Boost::program_options Boost::log pthread)
